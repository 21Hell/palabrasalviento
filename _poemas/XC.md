autor: Carlos Wolf
parrafos:
  - texto: |
      De punta a punta
      quiero describirte.
    alineamiento: izquierda
  - texto: |
      Describirte,
      no desescribirte,
      como escriba
      dar registro de tu pasear.
    alineamiento: izquierda
  - texto: |
      Resumirte en un verso,
      en una oración,
      una palabra.
    alineamiento: izquierda
  - texto: |
      Podría extenderme,
      hacerte novela,
      drama o romance,
      mejor aún,
      hacerte un poemario.
    alineamiento: derecha
  - texto: |
      Quisiera, escribirte.
    alineamiento: derecha
